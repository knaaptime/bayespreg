    "# JGSY2024 Replication with bayespecon\n",
    "\n",
    "This notebook replicates the core panel-SDM analysis in [Elhorst, J. P., Tziolas, I., Tan, C., & Milionis, P. (2024)](https://link.springer.com/article/10.1007/s10109-024-00440-5) using the replication files from [spatial-panels](https://spatial-panels.com/?page_id=2#1731681827580-52fa338b-6069), then compares the original outputs with `bayespecon` and provides a detailed side-by-side comparison to the published MATLAB outputs (Table 1, columns [1] and [2] from `JGSY2024_MainResults.m`).\n",
    "\n",
    "Replication targets in this notebook:\n",
    "- **Spec A**: 6-nearest-neighbor row-normalized weights\n",
    "- **Spec B**: exponential-distance row-normalized weights with fixed decay `exp(-distance/100)`\n",
    "\n",
    "Important scope notes:\n",
    "- JGSY2024 original estimation uses MATLAB routines (`sar_panelE_FE`) with classical asymptotics.\n",
    "- This notebook uses **Bayesian** estimation (`bayespecon.SDMPanelFE`) with posterior means for comparison.\n",
    "- We mirror two-way FE behavior via `model=3` (unit and time demeaning), which is equivalent to including unit FE and time dummies in balanced panels.\n",
    "- Parameterized-weight robustness specifications from `parameterized_W` are not estimated here because those routines are not currently part of `bayespecon`."

    "# Reproduce data construction from JGSY2024_MainResults.m\n",

    "def graph_from_dense(W_np: np.ndarray) -> Graph:\n",
    "    rows, cols = np.where(W_np > 0)\n",
    "    vals = W_np[rows, cols].astype(float)\n",
    "    return Graph.from_arrays(rows, cols, vals)\n",
    "\n",
    "\n",
    "def build_knn6_weights(distance_matrix: np.ndarray, m: int = 6) -> np.ndarray:\n",
    "    Dm = distance_matrix.copy().astype(float)\n",
    "    np.fill_diagonal(Dm, 0.0)\n",
    "    order = np.argsort(Dm, axis=1)\n",
    "    kth = np.take_along_axis(Dm, order[:, [m]], axis=1).ravel()\n",
    "\n",
    "    W = np.zeros_like(Dm, dtype=float)\n",
    "    for i in range(Dm.shape[0]):\n",
    "        mask = (Dm[i] > 0) & (Dm[i] <= kth[i])\n",
    "        W[i, mask] = 1.0\n",
    "\n",
    "    row_sum = W.sum(axis=1, keepdims=True)\n",
    "    row_sum[row_sum == 0] = 1.0\n",
    "    return W / row_sum\n",
    "\n",
    "\n",
    "def build_ed_weights(distance_matrix: np.ndarray) -> np.ndarray:\n",
    "    DD0 = distance_matrix.astype(float) / 100.0\n",
    "    W = np.exp(-DD0)\n",
    "    np.fill_diagonal(W, 0.0)\n",
    "    row_sum = W.sum(axis=1, keepdims=True)\n",
    "    row_sum[row_sum == 0] = 1.0\n",
    "    return W / row_sum\n",
    "\n",
    "\n",
    "W_knn6 = build_knn6_weights(D, m=6)\n",
    "W_ed = build_ed_weights(D)\n",
    "\n",
    "print(\"W_knn6 row sums:\", np.min(W_knn6.sum(axis=1)), np.max(W_knn6.sum(axis=1)))\n",
    "print(\"W_ed row sums:\", np.min(W_ed.sum(axis=1)), np.max(W_ed.sum(axis=1)))"

    "def fit_sdm_fe(y_vec, X_df, W_np, draws=600, tune=600, chains=1, seed=42):\n",
    "    model = SDMPanelFE(\n",
    "        y=y_vec,\n",
    "        X=X_df,\n",
    "        W=graph_from_dense(W_np),\n",
    "        N=N,\n",
    "        T=T,\n",
    "        model=3,\n",
    "        logdet_method=\"auto\",\n",
    "    )\n",
    "\n",
    "    idata = model.fit(\n",
    "        draws=draws,\n",
    "        tune=tune,\n",
    "        chains=chains,\n",
    "        cores=1,\n",
    "        target_accept=0.9,\n",
    "        random_seed=seed,\n",
    "        progressbar=False,\n",
    "        idata_kwargs={\"log_likelihood\": True},\n",
    "    )\n",
    "    return model, idata\n",
    "\n",
    "\n",
    "model_knn6, idata_knn6 = fit_sdm_fe(y, X_core, W_knn6)\n",
    "model_ed, idata_ed = fit_sdm_fe(y, X_core, W_ed)\n",
    "\n",
    "print(\"Fit complete for both specifications.\")"
