    "def extract_spreg_params(model):\n",
    "    names = list(getattr(model, \"name_x\", []))\n",
    "    betas = np.asarray(model.betas).flatten()\n",
    "    if len(betas) == len(names) + 1 and type(model).__name__ == \"GM_Lag\":\n",
    "        names = names + [\"rho\"]\n",
    "    if len(names) != len(betas):\n",
    "        names = [f\"param_{i}\" for i in range(len(betas))]\n",
    "    return pd.Series(betas, index=names, dtype=float)\n",
    "\n",
    "\n",
    "def extract_bayespecon_params(model_name, model, idata):\n",
    "    out = {}\n",
    "    beta_mean = idata.posterior[\"beta\"].mean((\"chain\", \"draw\")).values\n",
    "    beta_names = model._beta_names() if model_name in {\"SLX\", \"SDM\", \"SDEM\"} else list(model._feature_names)\n",
    "    for name, val in zip(beta_names, beta_mean):\n",
    "        out[name] = float(val)\n",
    "    if model_name in {\"SAR\", \"SDM\"}:\n",
    "        out[\"rho\"] = float(idata.posterior[\"rho\"].mean())\n",
    "    if model_name in {\"SEM\", \"SDEM\"}:\n",
    "        out[\"lambda\"] = float(idata.posterior[\"lam\"].mean())\n",
    "    harmonized = {}\n",
    "    for k, v in out.items():\n",
    "        key = k.replace(\"W*\", \"W_\")\n",
    "        if key.lower() == \"intercept\":\n",
    "            key = \"CONSTANT\"\n",
    "        harmonized[key] = v\n",
    "    return pd.Series(harmonized, dtype=float)\n",
    "\n",
    "\n",
    "def bayes_effects_df(model):\n",
    "    eff = model.spatial_effects()\n",
    "    return pd.DataFrame(\n",
    "        {\n",
    "            \"feature\": eff[\"feature_names\"],\n",
    "            \"direct_bayespecon\": eff[\"direct\"],\n",
    "            \"indirect_bayespecon\": eff[\"indirect\"],\n",
    "            \"total_bayespecon\": eff[\"total\"],\n",
    "        }\n",
    "    )\n",
    "\n",
    "\n",
    "def spreg_effects_df(model_name, sp_model):\n",
    "    if model_name == \"SEM\":\n",
    "        sp_params = extract_spreg_params(sp_model)\n",
    "        features = [f for f in [\"CONSTANT\", \"INC\", \"CRIME\"] if f in sp_params.index]\n",
    "        return pd.DataFrame(\n",
    "            {\n",
    "                \"feature\": features,\n",
    "                \"direct_spreg\": [float(sp_params[f]) for f in features],\n",
    "                \"indirect_spreg\": [0.0 for _ in features],\n",
    "                \"total_spreg\": [float(sp_params[f]) for f in features],\n",
    "            }\n",
    "        )\n",
    "\n",
    "    output = sp_model.output.copy()\n",
    "    output[\"regime\"] = \"global\"\n",
    "\n",
    "    def classify_var(var_name):\n",
    "        if var_name == \"CONSTANT\":\n",
    "            return \"o\"\n",
    "        if var_name in {\"lambda\", \"W_HOVAL\"}:\n",
    "            return \"rho\"\n",
    "        if var_name.startswith(\"W_\"):\n",
    "            return \"wx\"\n",
    "        return \"x\"\n",
    "\n",
    "    output[\"var_type\"] = output[\"var_names\"].map(classify_var)\n",
    "    sp_model.output = output\n",
    "\n",
    "    vars_x = output.query(\"var_type == 'x'\")\n",
    "    rho = float(np.squeeze(getattr(sp_model, \"rho\", 0.0))) if hasattr(sp_model, \"rho\") else 0.0\n",
    "    multipliers = spmultiplier(w, rho)\n",
    "    slx_lags = getattr(sp_model, \"slx_lags\", 0) or (1 if model_name in {\"SLX\", \"SDEM\"} else 0)\n",
    "    slx_vars = getattr(sp_model, \"slx_vars\", None)\n",
    "    if slx_vars is None:\n",
    "        slx_vars = \"All\"\n",
    "\n",
    "    total_spreg, direct_spreg, indirect_spreg = _sp_effects(\n",
    "        sp_model,\n",
    "        vars_x,\n",
    "        multipliers,\n",
    "        slx_lags=slx_lags,\n",
    "        slx_vars=slx_vars,\n",
    "    )\n",
    "\n",
    "    return pd.DataFrame(\n",
    "        {\n",
    "            \"feature\": vars_x[\"var_names\"].tolist(),\n",
    "            \"direct_spreg\": np.asarray(direct_spreg).flatten(),\n",
    "            \"indirect_spreg\": np.asarray(indirect_spreg).flatten(),\n",
    "            \"total_spreg\": np.asarray(total_spreg).flatten(),\n",
    "        }\n",
    "    )\n",
    "\n",
    "\n",
    "coef_rows = []\n",
    "effect_tables = {}\n",
    "\n",
    "for model_name in [\"SLX\", \"SAR\", \"SEM\", \"SDM\", \"SDEM\"]:\n",
    "    sp = extract_spreg_params(spreg_results[model_name])\n",
    "    by = extract_bayespecon_params(model_name, bayes_models[model_name], bayes_idata[model_name])\n",
    "\n",
    "    common = sorted(set(sp.index).intersection(set(by.index)))\n",
    "    for param in common:\n",
    "        coef_rows.append(\n",
    "            {\n",
    "                \"model\": model_name,\n",
    "                \"parameter\": param,\n",
    "                \"bayespecon_posterior_mean\": by[param],\n",
    "                \"spreg_estimate\": sp[param],\n",
    "                \"difference\": by[param] - sp[param],\n",
    "            }\n",
    "        )\n",
    "\n",
    "    beff = bayes_effects_df(bayes_models[model_name]).copy()\n",
    "    seff = spreg_effects_df(model_name, spreg_results[model_name]).copy()\n",
    "    merged = beff.merge(seff, on=\"feature\", how=\"inner\")\n",
    "    if not merged.empty:\n",
    "        merged[\"direct_difference\"] = merged[\"direct_bayespecon\"] - merged[\"direct_spreg\"]\n",
    "        merged[\"indirect_difference\"] = merged[\"indirect_bayespecon\"] - merged[\"indirect_spreg\"]\n",
    "        merged[\"total_difference\"] = merged[\"total_bayespecon\"] - merged[\"total_spreg\"]\n",
    "        merged.insert(0, \"model\", model_name)\n",
    "    effect_tables[model_name] = merged\n",
    "\n",
    "comparison = pd.DataFrame(coef_rows).sort_values([\"model\", \"parameter\"]).reset_index(drop=True)\n",
    "spatial_effects_comparison = pd.concat(\n",
    "    [tbl for tbl in effect_tables.values() if not tbl.empty],\n",
    "    ignore_index=True,\n",
    ")"
   ]

    "For effects, this notebook now uses `spreg`'s own spatial-impact utilities: `spreg.sputils.spmultiplier` and `spreg.sputils._sp_effects`. SEM is the one exception, since it has no spatial multiplier on the systematic component and therefore uses direct = coefficient, indirect = 0, total = coefficient."
