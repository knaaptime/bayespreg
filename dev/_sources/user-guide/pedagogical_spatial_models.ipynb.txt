    "## Conceptual Roadmap\n",
    "\n",
    "Let $W$ denote the spatial weights matrix and $WX$ the spatial lag of covariates.\n",
    "\n",
    "- **SLX**: $y = X\\beta + WX\\theta + \\epsilon$\n",
    "- **SAR**: $y = \\rho Wy + X\\beta + \\epsilon$\n",
    "- **SEM**: $y = X\\beta + u$, $u = \\lambda Wu + \\epsilon$\n",
    "- **SDM**: $y = \\rho Wy + X\\beta + WX\\theta + \\epsilon$\n",
    "- **SDEM**: $y = X\\beta + WX\\theta + u$, $u = \\lambda Wu + \\epsilon$\n",
    "\n",
    "In `bayespecon`, all models accept either:\n",
    "- formula mode: `model_class(formula=..., data=..., W=...)`, or\n",
    "- matrix mode: `model_class(y=..., X=..., W=...)`."
   ]

    "def fit_and_report(model_cls, formula, data, W, draws=400, tune=400, chains=4, seed=42):\n",
    "    model = model_cls(\n",
    "        formula=formula,\n",
    "        data=data,\n",
    "        W=W,\n",
    "        logdet_method=\"eigenvalue\",\n",
    "    )\n",
    "\n",
    "    idata = model.fit(\n",
    "        draws=draws,\n",
    "        tune=tune,\n",
    "        chains=chains,\n",
    "        target_accept=0.9,\n",
    "        random_seed=seed,\n",
    "        progressbar=False,\n",
    "        idata_kwargs={\"log_likelihood\": True},\n",
    "    )\n",
    "\n",
    "    summary = model.summary(round_to=3)\n",
    "    effects = model.spatial_effects()\n",
    "    effects_df = pd.DataFrame({\n",
    "        \"feature\": effects[\"feature_names\"],\n",
    "        \"direct\": effects[\"direct\"],\n",
    "        \"indirect\": effects[\"indirect\"],\n",
    "        \"total\": effects[\"total\"],\n",
    "    })\n",
    "\n",
    "    return model, idata, summary, effects_df"
   ]

    "def pull_scalar(idata, var_name):\n",
    "    if var_name not in idata.posterior:\n",
    "        return None\n",
    "    vals = idata.posterior[var_name].values.reshape(-1)\n",
    "    return {\n",
    "        \"mean\": float(np.mean(vals)),\n",
    "        \"sd\": float(np.std(vals)),\n",
    "        \"hdi_3%\": float(np.quantile(vals, 0.03)),\n",
    "        \"hdi_97%\": float(np.quantile(vals, 0.97)),\n",
    "    }\n",
    "\n",
    "rows = []\n",
    "for name, idata, var in [\n",

    "    (\"SDM\", idata_sdm, \"rho\"),\n",
    "    (\"SDEM\", idata_sdem, \"lam\"),\n",
    "]:\n",
    "    stats = pull_scalar(idata, var)\n",
